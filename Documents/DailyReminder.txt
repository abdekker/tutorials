dwStart := GetTickCount();
// Do something
dwElapsed := (GetTickCount() - dwStart);
Application.MessageBox(
	PAnsiChar(Format('Time=%dms', [dwElapsed])),
	'Developer Test', MB_ICONEXCLAMATION);	
Application.MessageBox(
	PAnsiChar(Format('Time1=%dms;time2=%dms', [dwElapsed1, dwElapsed2])),
	'Developer Test', MB_ICONEXCLAMATION);
Application.MessageBox(
	PAnsiChar(Format('Time1=%dms;time2=%dms;time3=%dms', [dwElapsed1, dwElapsed2, dwElapsed3])),
	'Developer Test', MB_ICONEXCLAMATION);
	
LomaX4.AddLogSystem(Format('Time=%dms', [dwElapsed]), False);

			dwElapsed1, dwElapsed2, dwElapsed3]), False);
